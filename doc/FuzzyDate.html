def initialize(year, month, day)
    if month.nil?
        @date_precision = :y
        super(year)
    else
        if day.nil?
            @date_precision = :m
        else
            @date_precision = :d
        end
    end
    @date_obj = Date.new(year, month||1, day||1)
    super(@date_obj)
end

def to_s
    case @date_precision
    when :y
        super.gsub(/-.*$/,'')
    when :m
        super.gsub(/-\d+$/,'')
    else
        super
    end
end
